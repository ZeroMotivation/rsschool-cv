    function pickPeaks(arr){
        let prevMax = -1;
        let pos = [];
        let peaks = [];
        let obj = {
            pos,
            peaks,
        }
        for(let i = 1; i <= arr.length - 1; i++) {
            if(arr[i] > arr[i - 1]) {
            prevMax = i;
            }
            
            if(arr[i] < arr[prevMax]) {
            pos.push(prevMax);
            peaks.push(arr[prevMax]);
            prevMax = -1;
            }
        }
        return obj;
    }
